```yml
# Var
package_list:
  - name: vim
    state: absent
  - name: nano
    state: latest
  - name: htop
  - name: tree
  - name: git
    state: latest
```

```yml
# Task
- name: Install required packages
  become: true
  ansible.builtin.package:
    name:
      - "{{ item.name }}"
    state: "{{ item.state | default('present') }}"
  loop: "{{ package_list }}"
  loop_control:
    label: "{{ item.name }} {{ item.state | default('present') }}"
```
